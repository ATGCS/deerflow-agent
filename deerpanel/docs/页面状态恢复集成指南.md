# 页面状态恢复集成指南

**更新时间**: 2026-04-05  
**适用页面**: chat.js, tasks.js

---

## 📋 概述

本指南说明如何在现有页面中集成三层状态恢复机制：

1. **Layer 1**: localStorage 快速恢复 (<100ms)
2. **Layer 2**: API 获取最新状态 (200-500ms)  
3. **Layer 3**: SSE 事件流实时更新 (即时)

---

## 🔧 Task 4.2: 聊天页面集成

### 修改 `src/pages/chat.js`

在 chat.js 的初始化部分添加状态恢复：

```javascript
// 在文件顶部添加导入
import { StateRestorationManager } from '../lib/state-persistence.js'
import { FloatingTaskPanel } from '../components/FloatingTaskPanel.js'
import { EmbeddedTaskDashboard } from '../components/EmbeddedTaskDashboard.js'
import { EventStreamManager } from '../lib/event-stream.js'

// 在页面初始化时添加状态恢复
export async function render() {
  const page = document.createElement('div')
  page.className = 'page'
  
  // ... 现有初始化代码 ...
  
  // ========== 新增：状态恢复 ==========
  
  // 创建状态恢复管理器
  const restorationManager = new StateRestorationManager()
  
  // 执行三层状态恢复
  await restorationManager.restore({
    // Layer 1: 从缓存恢复
    onCacheRestore: (data) => {
      console.log('[ChatPage] Layer 1: Restored from cache')
      if (data.tasks && data.tasks.length > 0) {
        // 恢复浮动面板状态
        if (data.panelState && data.panelState.isOpen) {
          const floatingPanel = new FloatingTaskPanel({
            position: data.panelState.position,
            size: data.panelState.size
          })
          floatingPanel.open()
          window.__floatingPanel = floatingPanel
        }
      }
    },
    
    // Layer 2: 从 API 恢复
    onAPIRestore: async (data) => {
      console.log('[ChatPage] Layer 2: Restored from API')
      if (data.tasks && data.tasks.length > 0) {
        // 检查是否需要显示嵌入式仪表板
        const executingTasks = data.tasks.filter(t => 
          t.status === 'executing' || t.status === 'planning'
        )
        
        if (executingTasks.length >= 2) {
          // 显示嵌入式仪表板
          const container = document.getElementById('embedded-dashboard-container')
          if (container) {
            const dashboard = new EmbeddedTaskDashboard(container)
            await dashboard.loadTasks()
            dashboard.startAutoRefresh()
            window.__embeddedDashboard = dashboard
          }
        }
      }
    },
    
    // Layer 3: SSE 连接
    onSSEConnect: () => {
      console.log('[ChatPage] Layer 3: SSE connected')
      // SSE 会自动通过事件流更新
    }
  })
  
  // ====================================
  
  return page
}
```

## 🔧 Task 4.2: 任务中心集成

### 修改 `src/pages/tasks.js`

在 tasks.js 的 loadTasks 函数中添加状态恢复：

```javascript
// 在文件顶部添加导入（已有 TaskConversationPanel）
import { StateRestorationManager } from '../lib/state-persistence.js'

// 修改 loadTasks 函数
async function loadTasks(page, state) {
  const container = page.querySelector('#tasks-list')
  const countEl = page.querySelector('#tasks-count')
  
  // ========== 新增：状态恢复 ==========
  
  // 第一次加载时执行状态恢复
  if (!window.__tasksPageRestored) {
    const restorationManager = new StateRestorationManager()
    
    await restorationManager.restore({
      onCacheRestore: (data) => {
        console.log('[TasksPage] Layer 1: Restored from cache')
        // 使用缓存数据快速渲染
        if (data.tasks) {
          state.tasks = data.tasks
          renderTasks(page, state)
        }
      },
      
      onAPIRestore: (data) => {
        console.log('[TasksPage] Layer 2: Restored from API')
        // 使用最新数据更新
        if (data.tasks) {
          state.tasks = data.tasks
          renderTasks(page, state)
        }
      },
      
      onSSEConnect: () => {
        console.log('[TasksPage] Layer 3: SSE connected')
        // 连接事件流（已在 FloatingTaskPanel 中处理）
      }
    })
    
    window.__tasksPageRestored = true
    return // 状态恢复完成后返回，避免重复加载
  }
  
  // ====================================
  
  // 正常加载流程（非首次加载）
  if (countEl) countEl.textContent = '加载中...'
  renderSkeleton(container)

  try {
    const tasks = await api.listAllTasks()
    state.tasks = tasks || []
    renderTasks(page, state)
  } catch (e) {
    container.innerHTML = `<div style="color:var(--error);padding:20px">加载失败：${escapeHtml(String(e))}</div>`
    if (countEl) countEl.textContent = '加载失败'
    toast('加载任务列表失败：' + e, 'error')
  }
}
```

## ⚠️ 注意事项

1. **缓存有效期**: 默认 5 分钟，超时后自动失效
2. **内存管理**: 页面卸载时要清理 SSE 连接和定时器
3. **并发更新**: 避免同时从多个来源更新同一数据
4. **错误处理**: 每层恢复失败不应影响其他层

## 📊 性能测试

### 性能指标测试

```javascript
// 1. 首次渲染时间
const start = performance.now()
// ... 执行渲染操作
const end = performance.now()
const renderTime = end - start
console.log('首次渲染时间:', renderTime, 'ms')
console.assert(renderTime < 1000, '首次渲染应该 < 1000ms')

// 2. 页面 FPS
let frameCount = 0
let lastTime = performance.now()

function measureFPS() {
  frameCount++
  const currentTime = performance.now()
  if (currentTime - lastTime >= 1000) {
    console.log('当前 FPS:', frameCount)
    console.assert(frameCount >= 50, 'FPS 应该 >= 50')
    frameCount = 0
    lastTime = currentTime
  }
  requestAnimationFrame(measureFPS)
}
measureFPS()

// 3. 内存占用
const memory = performance.memory
console.log('内存占用:', memory.usedJSHeapSize / 1048576, 'MB')
console.assert(memory.usedJSHeapSize < 104857600, '内存应该 < 100MB')

// 4. API 响应时间
const apiStart = performance.now()
await tasksAPI.listTasks()
const apiEnd = performance.now()
console.log('API 响应时间:', apiEnd - apiStart, 'ms')
console.assert(apiEnd - apiStart < 500, 'API 响应应该 < 500ms')
```

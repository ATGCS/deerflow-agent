## 1. 项目概述

### 1.1 项目背景

DeerFlow 是一个功能强大的多智能体系统，但现有的 Web 前端更偏向开发者使用。为了降低使用门槛，提升日常工作效率，需要开发一个**轻量化的桌面客户端**。

### 1.2 设计目标

| 目标 | 说明 |
|-----|------|
| **任务可视化** | 复杂任务的拆解、规划、执行全过程可视化 |
| **技能生态** | 完整的技能管理 + 技能市场，扩展系统能力 |
| **高效执行** | 快捷键唤起、拖拽操作、后台运行 |
| **低门槛** | 普通用户无需配置即可使用 |

### 1.3 核心价值

```
传统方式                      DeerFlow Desktop
─────────────────────         ─────────────────────────────────────────
打开浏览器 → 输入网址          按 Cmd+Shift+D 全局唤起
↓                            ↓
配置Agent → 创建对话           直接输入任务目标
↓                            ↓
黑盒等待AI执行                 清晰的任务规划 + 可视化执行
↓                            ↓
不知道进度，只能等待           实时进度 + 预计时间 + 系统通知
```

### 2.1 目标用户

- **项目经理**: 规划复杂任务，分配给AI执行，监控进度
- **开发者**: 快速启动代码相关任务，查看执行细节
- **普通用户**: 使用预设技能完成日常工作，无需配置

### 2.3 核心使用场景

#### 场景1: 快速启动任务
```
你在VSCode看代码 → 按 Cmd+Shift+D
→ 弹出桌面版小窗口
→ 输入"重构这段代码"
→ 自动关联当前文件
→ 后台运行，托盘显示进度
→ 完成后系统通知
```

#### 场景2: 复杂任务编排
```
你有一个大任务："重构整个项目"

1. AI规划阶段：
   - 自动拆解为10个子任务
   - 设置依赖关系
   - 预估时间

2. 编排阶段：
   - 分配不同Agent
   - 设置并行策略
   - 调整执行顺序

3. 执行阶段：
   - 可视化流程图
   - 实时进度条
   - Agent协作展示
```

#### 场景3: 技能扩展
```
你需要新能力："生成图表"

1. 打开技能市场
2. 搜索"chart" → 找到 chart_gen 技能
3. 一键安装
4. 立即在项目/任务中使用
```

#### 交互逻辑

1. **唤起方式**:
   - 全局快捷键: `Cmd/Ctrl + Shift + D`
   - 点击托盘图标
   - 右键托盘菜单

2. **快速输入**:
   - 输入文字 → 回车发送
   - 支持粘贴多行
   - 自动识别文件路径

3. **文件关联**:
   - 拖拽文件到窗口 → 自动上传并关联上下文
   - 支持文件夹拖拽

```typescript
// ============================================
// 项目 (Project)
// ============================================
interface Project {
  id: string;                    // 唯一标识
  name: string;                  // 项目名称
  description: string;           // 描述
  goal: string;                  // 项目目标

  // 状态
  status: 'planning' | 'queued' | 'running' | 'paused' | 'completed' | 'failed';

  // 关联
  tasks: Task[];                 // 任务列表

  // 执行配置
  config: ExecutionConfig;

  // 统计
  stats: {
    totalTasks: number;
    completedTasks: number;
    runningTasks: number;
    pendingTasks: number;
    failedTasks: number;
    totalDuration: number;       // 总耗时(秒)
  };

  // 时间
  createdAt: Date;
  updatedAt: Date;
  startedAt?: Date;
  completedAt?: Date;
}

// ============================================
// 任务 (Task)
// ============================================
interface Task {
  id: string;
  projectId: string;             // 所属项目
  name: string;                  // 任务名称
  description: string;           // 详细描述
  type: TaskType;

  // 执行配置
  agentName: string;             // 执行Agent
  autoAssign: boolean;           // 是否自动分配

  // 依赖关系
  dependencies: string[];        // 依赖的任务ID
  dependents: string[];          // 依赖于我的任务ID

  // 状态
  status: TaskStatus;
  progress: number;              // 0-100

  // 时间
  estimatedDuration: number;     // 预估时间(秒)
  actualDuration?: number;       // 实际用时
  createdAt: Date;
  startedAt?: Date;
  completedAt?: Date;

  // 输入/输出
  inputs: TaskInput[];
  outputs: TaskOutput[];

  // 子任务
  subtasks?: SubTask[];

  // 执行日志
  logs: TaskLogEntry[];

  // 错误信息
  error?: {
    message: string;
    code?: string;
    stack?: string;
  };
}

type TaskType =
  | 'analysis'      // 分析
  | 'refactor'      // 重构
  | 'write'         // 编写
  | 'review'        // 审查
  | 'test'          // 测试
  | 'deploy'        // 部署
  | 'custom';       // 自定义

type TaskStatus =
  | 'pending'       // 等待中
  | 'queued'        // 已排队
  | 'running'       // 运行中
  | 'paused'        // 已暂停
  | 'completed'     // 已完成
  | 'failed'        // 失败
  | 'cancelled';    // 已取消

// ============================================
// 技能 (Skill)
// ============================================
interface Skill {
  id: string;                    // 唯一标识 (如 "web_search")
  name: string;                  // 显示名称
  emoji?: string;                // 图标

  // 元信息
  description: string;
  version: string;
  author: string;
  authorAvatar?: string;
  license: string;
  tags: string[];
  category: SkillCategory;

  // 状态
  status: 'enabled' | 'disabled' | 'error';
  isBuiltIn: boolean;
  isFromMarket: boolean;

  // 内容
  skillMdContent: string;        // SKILL.md
  tools: ToolDefinition[];       // 工具定义

  // 配置
  configSchema?: ConfigSchema;
  config?: Record<string, unknown>;

  // 市场信息
  marketInfo?: {
    marketId: string;
    installSource: string;
    latestVersion?: string;
    updateAvailable?: boolean;
    downloads?: number;
    rating?: number;
  };

  // 时间
  installedAt: Date;
  updatedAt: Date;
}

type SkillCategory =
  | 'search'
  | 'code'
  | 'document'
  | 'test'
  | 'devops'
  | 'communication'
  | 'data'
  | 'chart'
  | 'custom';

// ============================================
// Agent
// ============================================
interface Agent {
  name: string;
  description?: string;
  model?: string;
  toolGroups?: string[];
  soul?: string;

  // 运行时状态
  runtimeStatus: 'idle' | 'busy' | 'offline';
  currentTaskId?: string;
  lastActiveAt?: Date;

  // 统计
  stats: {
    totalTasksCompleted: number;
    totalTasksFailed: number;
    averageTaskDuration: number;
  };
}

// ============================================
// 执行配置
// ============================================
interface ExecutionConfig {
  mode: 'sequential' | 'parallel' | 'smart';
  maxConcurrency: number;        // 最大并发Agent数
  autoRetry: boolean;
  maxRetries: number;
  continueOnError: boolean;
  timeout: number;               // 单个任务超时(秒)
}

// ============================================
// 技能市场
// ============================================
interface MarketSkill {
  id: string;
  name: string;
  description: string;
  version: string;
  author: string;
  authorAvatar?: string;
  emoji?: string;
  category: SkillCategory;
  tags: string[];

  stats: {
    downloads: number;
    rating: number;
    ratingCount: number;
  };

  isOfficial: boolean;
  isVerified: boolean;

  installUrl: string;
  size: number;

  dependencies?: {
    deerflowVersion?: string;
    skills?: string[];
    tools?: string[];
  };

  screenshots?: string[];
  readmeExcerpt?: string;
}
```

### 5.2 辅助类型

```typescript
// 任务输入
interface TaskInput {
  type: 'text' | 'file' | 'url' | 'context';
  content: string;
  metadata?: Record<string, unknown>;
}

// 任务输出
interface TaskOutput {
  type: 'file' | 'text' | 'artifact';
  name: string;
  path?: string;
  content?: string;
  size?: number;
}

// 任务日志
interface TaskLogEntry {
  timestamp: Date;
  level: 'info' | 'warn' | 'error' | 'debug';
  source: string;                // Agent名或系统
  message: string;
  data?: Record<string, unknown>;
}

// 子任务
interface SubTask {
  id: string;
  name: string;
  status: TaskStatus;
  progress: number;
  order: number;
}

// 工具定义
interface ToolDefinition {
  name: string;
  description: string;
  parameters: ParameterSchema;
}

// 配置项定义
interface ConfigSchema {
  [key: string]: {
    type: 'string' | 'number' | 'boolean' | 'select';
    description: string;
    default?: unknown;
    options?: string[];         // 用于select类型
    required?: boolean;
    secret?: boolean;           // 是否敏感(如API Key)
  };
}

// 依赖图
interface DependencyGraph {
  nodes: TaskNode[];
  edges: { from: string; to: string; type: 'strong' | 'weak' }[];
}

interface TaskNode {
  id: string;
  name: string;
  status: TaskStatus;
  agentName: string;
  progress: number;
  x: number;
  y: number;
}
```

## 7. UI/UX 设计规范

### 7.1 设计原则

1. **清晰优先**: 任务状态和进度必须一目了然
2. **层次分明**: 主Agent和子Agent的关系清晰展示
3. **实时反馈**: 操作后立即反馈，避免用户焦虑
4. **深色基调**: 科技感，减少眼部疲劳
5. **克制用色**: 仅用颜色表达状态，避免过度装饰
